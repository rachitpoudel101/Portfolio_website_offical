class Portfolio:
  def __init__(self):
    self.projects = []

  def featured_projects(self):
    return [project for project in self.projects if project.featured]

  def recent_projects(self, limit=3):
    return sorted(self.projects, key=lambda x: x.created_at, reverse=True)[:limit]
